@startuml
actor Main
participant StandardWebServer
participant DefaultControllerScanner
participant DefaultControllerMapper
participant Mapping
participant SunHttpServer

Main -> StandardWebServer : new()
StandardWebServer -> DefaultControllerScanner : scanner()
DefaultControllerScanner -> StandardWebServer : controllersScanned:List<Class<Controller>>
StandardWebServer -> DefaultControllerMapper : mapping(List<Class<Controller>>
DefaultControllerMapper -> Mapping : new()
Mapping -> DefaultControllerMapper : mapper : Mapping
DefaultControllerMapper -> StandardWebServer : mapper:Mapping
StandardWebServer -> SunHttpServer : addConfiguration()
SunHttpServer -> StandardWebServer : server configured
@enduml
